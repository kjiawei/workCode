void CreateNotice(INT16U x,INT16U y, ENOTICETYPE AlarmType, INT8U AlarmNum)

{

 DrawWindow(x, y, x+AAREAXX, y+AAREAYY, GUI_GREEN,0xc0c0c0);   ////绘3D窗口paint window 3D

 GUI_SetColor(GUI_RED);

 GUI_SetFont(&GUI_Font_HZ16);

 GUI_SetTextMode(GUI_TM_TRANS);

 switch(AlarmType)

   {

    case TYPE_ERR:    GUI_DispStringAt_UC("错误error ",x+6, y+4); GUI_DrawBitmap(&bmerror, x+FLAGX, y+FLAGY); break;

    case TYPE_ALARM:  GUI_DispStringAt_UC("警告alarm ",x+6, y+4); GUI_DrawBitmap(&bmalarm, x+FLAGX, y+FLAGY); break;

    case TYPE_ASK:    GUI_DispStringAt_UC("确认confirm ",x+6, y+4); GUI_DrawBitmap(&bmask, x+FLAGX, y+FLAGY); break;

    default:          GUI_DispStringAt_UC("错误error ",x+6, y+4); GUI_DrawBitmap(&bmerror, x+FLAGX, y+FLAGY); break;

   }

 GUI_SetColor(GUI_BLACK);  

 (*DisNotic[AlarmNum])(x+CHARX,y+CHARY);   

 if(AlarmType == TYPE_ASK)

   {

    ButtonYes = BUTTON_Create( x+ABUTTONX1, y+ABUTTONY , ABUTTONL, ABUTTONH,GUI_ID_YES ,BUTTON_CF_SHOW );

    ButtonNo = BUTTON_Create( x+ABUTTONX2,  y+ABUTTONY , ABUTTONL, ABUTTONH,GUI_ID_NO ,BUTTON_CF_SHOW );

    BUTTON_SetText(ButtonYes, "确定");confirm

    BUTTON_SetText(ButtonNo, "取消");cancel

   }

 else if(AlarmType == TYPE_ERR)

   {

    ButtonYes = BUTTON_Create( x+AAREAXX/2-ABUTTONL/2, y+ABUTTONY ,ABUTTONL, ABUTTONH,GUI_ID_NO ,BUTTON_CF_SHOW );

    BUTTON_SetText(ButtonYes, "确定");confirm

   }

 else if(AlarmType == TYPE_ALARM)

   {

    ButtonYes = BUTTON_Create( x+AAREAXX/2-ABUTTONL/2, y+ABUTTONY , ABUTTONL, ABUTTONH,GUI_ID_NO ,BUTTON_CF_SHOW );

    BUTTON_SetText(ButtonYes, "确定");confirm

   }    

 mG_NoticeFlag.Flag = YES;   

}
